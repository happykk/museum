# 简介 - 底部导航 # 安装 ``` js import 'mk-ui/nav.js' import 'mk-ui/nav.css' ``` # 默认 # mk-nav 组件 ## v-model=show 显示或隐藏 - type: boolean ## color (属性) 默认颜色 - type: string - value: '#bab4b3' ## selected-color (属性) 选中的颜色 - type: string - value: '#fe7773' # mk-nav-item 组件 ## icon-class (属性) 图标class名称 - type: string - value: 'iconfontcom' ## icon (属性) 图标码 - type: string - value: '图标编码' ## selected-icon (属性) 选中的图标 - type: string - value: '图标编码' ## icon-num (属性) 显示消息条数, 当大于99时显示99+ - type: number | string ## text (属性) 文本内容 - type: string ## dot (属性) 加点标记 - type: boolean ## img (属性) 图标图片路径 - type: string ## selected-img (属性) 选中后图片路径 - type: string ## selected (属性) 选中当前项 - type: boolean ## .$on('selected', ({ e, item }) => {}) 选中事件 e 事件对象 - type: Object item 当前选中的组件对象 - type: Object
